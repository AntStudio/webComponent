添加自定义按钮，并调用插件内置函数

代码


$("div.CustomButtonTable").table({
	columns:[
			{name:"id",render:function(rowData){return rowData.name;},width:200,display:'id(限宽200px)',sort:true},
			{name:"name",display:"权限名称(居中)",align:"center",sort:true,sortName:"name"},
			{name:"code"},
			{display:'附加列测试(居右)',align:"right",render:function(columnData){return "测试内容 for "+columnData.name;}}
			],
	data:{
		totalItemsCount:"18",
		items:[
			{"id":1,"name":"系统设置","code":"000001"},
			{"id":2,"name":"菜单管理","code":"000002"},
			{"id":3,"name":"菜单信息","code":"000003"},
			{"id":4,"name":"用户权限","code":"000004"},
			{"id":6,"name":"用户信息","code":"000005"},
			{"id":5,"name":"用户列表","code":"000006"}
			]
	},
	title: "CustomButtonTable",
	buttons:[
			{
			text : "统计选中行数",//按钮文字
			click : function(btnOpts){
				var tableInstance = this;
				alert("一共选择了 "+tableInstance.getSelect().length+"行");
				}//点击回调，上下文为table,btnOpts代表当前button的配置
			},
			{
			text : "刷新表格",
			click : function(btnOpts){
				var tableInstance = this;
				tableInstance.refresh();
				alert("表格已刷新");
				}
			},
			{
			text : "导出json数据",
			click : function(btnOpts){
				var tableInstance = this;
				console.log(tableInstance.getChangedRows());
				alert("到控制台查看数据内容。。。");
				}
			}
			]
});
